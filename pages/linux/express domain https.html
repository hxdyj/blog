# 配置 nginx

server {
   listen 80;
   listen 443 ssl http2; //宝塔必须有这个
   server_name  xxx;
   # 开启 ssl
   ssl on;
   # 指定 ssl 证书路径
   ssl_certificate /www/wwwroot/xxx/xxx/https/server.pem;
   # 指定私钥文件路径
   ssl_certificate_key /www/wwwroot/xxx/xxx/https/server.key;

   location / {
       proxy_pass      https://127.0.0.1:3000;//指定反向代理
       }
   }
